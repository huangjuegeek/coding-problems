import java.util.HashMap;
import java.util.Map;
import java.util.Scanner;

public class Main {
	static Scanner sc = new Scanner(System.in);

	public static void main(String[] args) {
		int n = sc.nextInt();
		Trie trie = new Trie();
		String str = "";
		while (n-- != 0) {
			str = sc.next();
			trie.insert(str);
		}
		int m = sc.nextInt();
		while (m-- != 0) {
			str = sc.next();
			System.out.println(trie.getCount(str));
		}
	}

	public static class Trie {

		private TrieNode root;

		public Trie() {
			this.root = new TrieNode(null);
		}

		public void insert(String vocabulary) {
			int len = vocabulary.length();
			TrieNode pre = root, cur = null;
			for (int i = 0; i < len; ++i) {
				cur = pre.set(vocabulary.charAt(i));
				cur.CountAdd();
				pre = cur;
			}
		}

		public int getCount(String vocabulary) {
			int len = vocabulary.length();
			TrieNode pre = root, cur = null;
			for (int i = 0; i < len; i++) {
				cur = pre.get(vocabulary.charAt(i));
				if (cur == null) {
					return 0;
				}
				pre = cur;
			}
			return cur.getCount();
		}

		public static class TrieNode {
			Map<Character, TrieNode> map;
			private int count = 0;
			private Character value = 0;
			public static int SIZE = 30;

			public TrieNode(Character ch) {
				this.value = ch;
				map = new HashMap<>(SIZE);
			}

			public TrieNode get(Character ch) {
				return map.get(ch);
			}

			public TrieNode set(Character ch) {
				TrieNode t = map.get(ch);
				if (t == null) {
					t = new TrieNode(ch);
					this.map.put(ch, t);
				}
				return t;
			}

			public int getCount() {
				return this.count;
			}

			public Character getValue() {
				return this.value;
			}

			public void CountAdd() {
				this.count++;
			}
		}
	}
}

import java.util.Map;
import java.util.Scanner;
import java.util.TreeMap;

public class Main {
	private static class Trie {
		public int count;
		public Map<Character, Trie> children;

		public Trie() {
			this.count = 0;
			this.children = new TreeMap<Character, Trie>();
		}

		public void insert(String word) {
			Trie trie = this;
			trie.count++;
			for (int i = 0; i < word.length(); i++) {
				char ch = word.charAt(i);
				Trie node = trie.children.get(ch);
				if(node == null) {
					node = new Trie();
					trie.children.put(ch, node);
				}
				node.count++;
				trie = node;
			}
		}

		public int getCountByPrefix(String prefix) {
			Trie trie = this;
			for (int i = 0; i < prefix.length(); i++) {
				char ch = prefix.charAt(i);
				Trie node = trie.children.get(ch);
				if(node == null) {
					return 0;
				}
				trie = node;
			}
			return trie.count;
		}
	}

	static Scanner scanner = new Scanner(System.in);
	
	public static void main(String[] args) {
		Trie trie = new Trie();
		int n = scanner.nextInt();
		for (int i = 0; i < n; i++) {
			String word = scanner.next();
			trie.insert(word);
		}

		int m = scanner.nextInt();
		for (int i = 0; i < m; i++) {
			String prefix = scanner.next();
			System.out.println(trie.getCountByPrefix(prefix));
		}
	}
}
