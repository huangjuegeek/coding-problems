# 200. Number of Islands
> https://leetcode.com/problems/number-of-islands

## Solution 1: DFS
```java
/*
从左上到右下，遍历grid中的每一个格子，遇到'1'时，说明存在一个岛，计数count加1
然后使用深度优先遍历(DFS)，找到属于该岛的所有'1'，将其置为'#'
当一个岛的所有'1'都被置为'#'时，对该岛的DFS完成
完成剩下格子的遍历，直到找到所有的岛
*/
public class Solution {
    public int numIslands(char[][] grid) {
        if (grid == null || grid.length == 0)
            return 0;
        int count = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == '1') {
                    count++;
                    dfs(grid, i, j);
                }
            }
        }
        return count;
    }

    private void dfs(char[][] grid, int m, int n) {
        if (m < 0 || m >= grid.length || n < 0 || n >= grid[0].length || grid[m][n] != '1') {
            return;
        }
        grid[m][n] = '#';
        dfs(grid, m + 1, n);
        dfs(grid, m - 1, n);
        dfs(grid, m, n + 1);
        dfs(grid, m, n - 1);
    }
}
```

```java
/*
若不希望改变grid中元素的值，则需要使用一个二维数组visited来标记已访问过的格子
*/
class Solution {
    public int numIslands(char[][] grid) {
        if (grid == null || grid.length == 0)
            return 0;
        int count = 0, m = grid.length, n = grid[0].length;
        boolean[][] visited = new boolean[m][n];
        
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (visited[i][j]) {
                    continue;
                }
                if (grid[i][j] == '1') {
                    count++;
                    dfs(grid, visited, i, j);
                }
            }
        }
        
        return count;
    }

    private void dfs(char[][] grid, boolean[][] visited, int m, int n) {
        if (m < 0 || m >= grid.length || n < 0 || n >= grid[0].length || grid[m][n] != '1' || visited[m][n]) {
            return;
        }
        visited[m][n] = true;
        dfs(grid, visited, m + 1, n);
        dfs(grid, visited, m - 1, n);
        dfs(grid, visited, m, n + 1);
        dfs(grid, visited, m, n - 1);
    }
}
```

## Solution 2: Union Find 并查集
```java
public class Solution {
    public int numIslands(char[][] grid) {
        if (grid.length == 0 || grid[0].length == 0)
            return 0;
        int m = grid.length, n = grid[0].length;
        UnionFind uf = new UnionFind(m, n, grid);

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == '0') {
                    continue;
                }
                int p = i * n + j;
                int q;

                // 没有必要考虑 上 方向
                // if(i > 0 && grid[i - 1][j] == '1') {
                // q = p - n;
                // uf.union(p, q);
                // }

                // 下 方向
                if (i < m - 1 && grid[i + 1][j] == '1') {
                    q = p + n;
                    uf.union(p, q);
                }

                // 没有必要考虑 左 方向
                // if(j > 0 && grid[i][j - 1] == '1') {
                // q = p - 1;
                // uf.union(p, q);
                // }

                // 右 方向
                if (j < n - 1 && grid[i][j + 1] == '1') {
                    q = p + 1;
                    uf.union(p, q);
                }
            }
        }

        return uf.count;
    }
}

class UnionFind {
    public int count = 0;      // 组的个数
    public int[] id = null;    // 组号
    public int[] size = null;  // 每个组的大小

    public UnionFind(int m, int n, char[][] grid) {
        // 先遍历一遍grid，初始状态，有多少个'1'，就有多少个组
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == '1') {
                    count++;
                }
            }
        }
        id = new int[m * n];
        size = new int[m * n];
        for (int i = 0; i < m * n; i++) {
            id[i] = i;
            size[i] = 1;
        }
    }

    public int find(int p) {
        while (p != id[p]) {  // 满足该条件意味着p已经被并入了其他组里，那么沿着p的当前id在id数组里找到p所属组的id
            id[p] = id[id[p]];
            p = id[p];
        }
        return p;
    }

    public boolean isConnected(int p, int q) {
        return find(p) == find(q);
    }

    public void union(int p, int q) {
        int pId = find(p);             // p所属组的id
        int qId = find(q);             // q所属组的id
        if (pId == qId) {
            return;
        }
        if (size[pId] > size[qId]) {    // 将q并入p所属的组
            id[qId] = pId;
            size[pId] += size[qId];
        } else {                      // 将p并入q所属的组
            id[pId] = qId;
            size[qId] += size[pId];
        }
        count--;
    }
}
```
