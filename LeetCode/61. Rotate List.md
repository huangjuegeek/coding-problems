# 61. Rotate List
> https://leetcode.com/problems/rotate-list

## Solution 1:
```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode rotateRight(ListNode head, int k) {
        if (head == null || k == 0) {
            return head;
        }
        
        int len = 0;    // 链表长度len
        for (ListNode cur = head; cur != null; cur = cur.next) {
            len++;
        }
        
        k %= len;
        if (k == 0) {    // 不需要rotate，直接返回head
            return head;
        }
        
        ListNode fast = head, slow = head;
        for (int i = 0; i < k; i++) {    // fast先走k(已对len取模)步
            fast = fast.next;
        }
        
        // fast和slow一起走，直到fast指向整个链表最后一个结点
        // 此时，slow将链表分为两部分。slow指向rotate前，左半边链表的尾结点；fast指向rotate前，右半边链表的尾结点。
        while (fast.next != null) {    
            fast = fast.next;
            slow = slow.next;
        }
        
        ListNode newHead = slow.next;
        slow.next = null;
        fast.next = head;
        
        return newHead;
    }
}
```

## Solution 2:
```java
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode rotateRight(ListNode head, int k) {
        if (head == null || k == 0) {
            return head;
        }
        
        int len = 0;    // 链表长度len
        for (ListNode cur = head; cur != null; cur = cur.next) {    
            len++;
        }
        
        k %= len;
        if (k == 0) {    // 不需要rotate，直接返回head
            return head;
        }
        
        // leftLast: rotate前，左半边链表的尾结点；rightFirst: rotate前，右半边链表的头结点；rightLast: 右半边链表的尾结点
        ListNode leftLast = head, rightFirst = head, rightLast = head, cur = head;
        for (int i = 1; i <= len; i++, cur = cur.next) {
            if (i == len - k) {
                leftLast = cur;
            }
            if (i == len) {
                rightLast = cur;
            }
        }
        
        ListNode newHead = leftLast.next;
        leftLast.next = null;
        rightLast.next = head;
        
        return newHead;
    }
}
```
